function refresh() { // 1. 初始化钱包：确保所有成员即使没数据也出现在 Wallet 里 const wallet = team.reduce((acc, n) => { acc[n] = { balance: 0, progress: 0 }; return acc; }, {}); const redeemed = {}; const attendanceRows = []; // 2. 清洗数据并分类 csvData.forEach(originalRow => { // 去除每一项多余的空格 const row = originalRow.map(item => item.trim()); if (row[0] === "REDEEM") { redeemed[row[1]] = true; } else if (row[0] === "TRANSFER") { // 转账逻辑：确保名字匹配（忽略大小写差异） const from = team.find(n => n.toLowerCase() === row[1].toLowerCase()); const to = team.find(n => n.toLowerCase() === row[2].toLowerCase()); if (from) wallet[from].balance--; if (to) wallet[to].balance++; } else if (row[0] && row[0].includes('-')) { // 打卡逻辑 attendanceRows.push(row); team.forEach((name, i) => { const status = row[i+1]; if (status === "On Time" || status === "Remote") { wallet[name].progress++; if (wallet[name].progress >= 10) { wallet[name].balance++; wallet[name].progress = 0; } } else if (status === "Golden Ticket") { wallet[name].balance--; } }); } }); walletState = wallet; // 更新全局状态 // 3. 计算 Streak (保持跨月连续) const sorted = attendanceRows.sort((a, b) => parseLocalDate(a[0]) - parseLocalDate(b[0])); let tempStreak = 0; sorted.forEach(row => { const d = parseLocalDate(row[0]); if (d.getDay() !== 0 && d.getDay() !== 6) { const hasLate = row.slice(1).includes("Late"); tempStreak = hasLate ? 0 : tempStreak + 1; } }); currentStreak = tempStreak; document.getElementById('streakDisplay').innerText = currentStreak; // 4. 渲染 Wallet UI (确保一定有内容渲染) const walletGrid = document.getElementById('walletGrid'); walletGrid.innerHTML = team.map(n => `
${n}
${wallet[n].balance}🎫
${wallet[n].progress}/10 to next
`).join(''); // 5. 渲染里程碑进度栏 document.getElementById('milestoneBar').innerHTML = milestones.map(m => { const isDone = currentStreak >= m.d; return `
${m.icon} ${m.t} ${isDone ? m.d : currentStreak}/${m.d}
`; }).join(''); renderCalendar(); }
